Archivo compilado: 

Get-ChildItem -Recurse -File |
Where-Object {
    $_.FullName -notmatch "node_modules" -and
    $_.FullName -notmatch "\\icons\\" -and
    $_.FullName -notmatch "\\.env$"
} |
ForEach-Object {
    Add-Content -Path cooptrack_snapshot.txt -Value "===============================" -Encoding utf8
    Add-Content -Path cooptrack_snapshot.txt -Value "FILE: $($_.FullName)" -Encoding utf8
    Add-Content -Path cooptrack_snapshot.txt -Value "===============================" -Encoding utf8

    Get-Content $_.FullName -Raw | Add-Content -Path cooptrack_snapshot.txt -Encoding utf8

    Add-Content -Path cooptrack_snapshot.txt -Value "`n" -Encoding utf8
}
